Our first template

  • Move index.html file into index.hbs
  • Install 2 new plugins gulp-compile-handlebars and gulp-rename
  • Require those 2 plugins in you gulpfile
  • Create and new gulp task called templates 
                
            gulp.task(‘templates’, function(){
              return gulp.src(['src/templates/**/*.hbs']) //watch our template dir
                .pipe(handlebars()) //pipe through handlebars
                .pipe(rename(function(path){ //rename, take a function
                path.extname=’.html’
                })
                .pipe(gulp.dest(‘./’));
Fix watch, we wanted to run all these tasks when gulp fires up 
                
gulp.task('default', ['templates', 'scripts', 'styles', 'images'],
  function() {
    browserSync.init({
      server: './'
    });

    gulp.watch('src/scripts/**/*.js', ['scripts']);
    gulp.watch('src/styles**/*.css', ['styles']);
    gulp.watch('src/img/**/*', ['images']);
    gulp.watch('src/templates/**/*.hbs', ['templates']);
    gulp.watch('*.html', browserSync.reload);
